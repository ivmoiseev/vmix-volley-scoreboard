/**
 * –í—ã—á–∏—Å–ª—è–µ—Ç –ø—Ä–æ–¥–æ–ª–∂–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å –ø–∞—Ä—Ç–∏–∏ –≤ –º–∏–Ω—É—Ç–∞—Ö
 * @param {number} startTime - timestamp –Ω–∞—á–∞–ª–∞ –ø–∞—Ä—Ç–∏–∏ –≤ –º–∏–ª–ª–∏—Å–µ–∫—É–Ω–¥–∞—Ö
 * @param {number} endTime - timestamp –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è –ø–∞—Ä—Ç–∏–∏ –≤ –º–∏–ª–ª–∏—Å–µ–∫—É–Ω–¥–∞—Ö
 * @returns {number} –ø—Ä–æ–¥–æ–ª–∂–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å –≤ –º–∏–Ω—É—Ç–∞—Ö (–æ–∫—Ä—É–≥–ª–µ–Ω–∏–µ –≤–Ω–∏–∑)
 */
function calculateSetDuration(startTime, endTime) {
  if (!startTime || !endTime) {
    return 0;
  }
  const durationMs = endTime - startTime;
  const durationMinutes = Math.floor(durationMs / (60 * 1000));
  return durationMinutes;
}

/**
 * –ü–æ–ª—É—á–∞–µ—Ç –∑–∞–≤–µ—Ä—à–µ–Ω–Ω—ã–µ –ø–∞—Ä—Ç–∏–∏ –¥–æ —É–∫–∞–∑–∞–Ω–Ω–æ–≥–æ –Ω–æ–º–µ—Ä–∞ –≤–∫–ª—é—á–∏—Ç–µ–ª—å–Ω–æ
 * @param {Array} sets - –º–∞—Å—Å–∏–≤ –≤—Å–µ—Ö –ø–∞—Ä—Ç–∏–π
 * @param {number} maxSetNumber - –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π –Ω–æ–º–µ—Ä –ø–∞—Ä—Ç–∏–∏ (–≤–∫–ª—é—á–∏—Ç–µ–ª—å–Ω–æ)
 * @returns {Array} –º–∞—Å—Å–∏–≤ –∑–∞–≤–µ—Ä—à–µ–Ω–Ω—ã—Ö –ø–∞—Ä—Ç–∏–π
 */
function getCompletedSetsUpTo(sets, maxSetNumber) {
  if (!sets || !Array.isArray(sets)) {
    return [];
  }
  return sets
    .filter(
      (set) =>
        set.setNumber <= maxSetNumber &&
        (set.status === 'completed' || set.completed === true)
    )
    .sort((a, b) => a.setNumber - b.setNumber);
}

/**
 * –í—ã—á–∏—Å–ª—è–µ—Ç —Å—á–µ—Ç –ø–æ —Å–µ—Ç–∞–º –¥–ª—è –∫–æ–º–∞–Ω–¥—ã
 * @param {Array} sets - –º–∞—Å—Å–∏–≤ –∑–∞–≤–µ—Ä—à–µ–Ω–Ω—ã—Ö –ø–∞—Ä—Ç–∏–π
 * @param {string} team - 'A' –∏–ª–∏ 'B'
 * @returns {number} –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –≤—ã–∏–≥—Ä–∞–Ω–Ω—ã—Ö –ø–∞—Ä—Ç–∏–π
 */
function calculateSetsScore(sets, team) {
  if (!sets || !Array.isArray(sets)) {
    return 0;
  }
  return sets.filter((set) => {
    return team === 'A' ? set.scoreA > set.scoreB : set.scoreB > set.scoreA;
  }).length;
}

/**

 * @param {Object} match - –¥–∞–Ω–Ω—ã–µ –º–∞—Ç—á–∞

function formatSetScoreInputData(match, setNumber) {
  if (!match) {
    return { fields: {} };
  }

  const fields = {};

  // –û–±—â–∏–µ –ø–æ–ª—è
  fields['TeamA'] = match.teamA?.name || '';
  fields['TeamB'] = match.teamB?.name || '';

  // –ü–æ–ª—É—á–∞–µ–º –∑–∞–≤–µ—Ä—à–µ–Ω–Ω—ã–µ –ø–∞—Ä—Ç–∏–∏ –¥–æ —É–∫–∞–∑–∞–Ω–Ω–æ–≥–æ –Ω–æ–º–µ—Ä–∞

  const scoreASets = calculateSetsScore(completedSets, 'A');
  const scoreBSets = calculateSetsScore(completedSets, 'B');
  fields['ScoreASets'] = String(scoreASets);
  fields['ScoreBSets'] = String(scoreBSets);

  // –ü–æ–ª—è –¥–ª—è –∫–∞–∂–¥–æ–π –∑–∞–≤–µ—Ä—à–µ–Ω–Ω–æ–π –ø–∞—Ä—Ç–∏–∏
  completedSets.forEach((set) => {
    const duration = calculateSetDuration(set.startTime, set.endTime);
    const setNum = set.setNumber;

    // –ò—Å–ø–æ–ª—å–∑—É–µ–º –ø—É—Å—Ç—É—é —Å—Ç—Ä–æ–∫—É –¥–ª—è duration, –µ—Å–ª–∏ –≤—Ä–µ–º—è –æ—Ç—Å—É—Ç—Å—Ç–≤—É–µ—Ç
    fields[`Set${setNum}Duration`] = duration > 0 ? String(duration) : '';
    fields[`Set${setNum}ScoreA`] = String(set.scoreA || 0);
    fields[`Set${setNum}ScoreB`] = String(set.scoreB || 0);
  });

  // –î–ª—è –ø–∞—Ä—Ç–∏–π, –∫–æ—Ç–æ—Ä—ã–µ –µ—â–µ –Ω–µ –∑–∞–≤–µ—Ä—à–µ–Ω—ã (–¥–æ setNumber), –æ—Ç–ø—Ä–∞–≤–ª—è–µ–º –ø—É—Å—Ç—ã–µ —Å—Ç—Ä–æ–∫–∏
  for (let i = 1; i <= setNumber; i++) {

    if (!setExists) {

      fields[`Set${i}Duration`] = '';
      fields[`Set${i}ScoreA`] = '';
      fields[`Set${i}ScoreB`] = '';
    }
  }

  return { fields };
}
